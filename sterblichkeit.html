import matplotlib.pyplot as plt
import numpy as np
from math import exp
%matplotlib inline

INFEKTIONSRATE = 1.0
# Wie hoch ist die Wahrscheinlichkeit, nach einer Infektion zu versterben
STERBE_RISIKO = 1.0
# Wie lange dauert es, bis ein Infizierter stirbt
STERBE_VERZGERUNG = 20.0
# Begin der ansteckungsfreien Phase gab
KONTROLLPHASE_START = 40.0
# Ende der ansteckungsfreien Phase
KONTROLLPHASE_ENDE = 70.0

# Anzahl infizierter zum Zeitpunkt(t). 
def infektionen(t):
    if t <= KONTROLLPHASE_START:
        return t * INFEKTIONSRATE
    elif t > KONTROLLPHASE_START and t < KONTROLLPHASE_ENDE:
        return infektionen(KONTROLLPHASE_START)
    else:
        kontrollzeit = KONTROLLPHASE_ENDE - KONTROLLPHASE_START 
        return t - kontrollzeit
# Todesflle zum Zeitpunkt (t)
def tote(t):
    return max(0.0,infektionen(t-STERBE_VERZGERUNG) * STERBE_RISIKO)

def fallsterblichkeit(t):
    return tote(t)/ infektionen(t)

t = np.arange(0.01, 100.0, 0.01)
data1 = [infektionen(_) for _ in t ]
data2 = [tote(x) for x in t]
data3 = [fallsterblichkeit(x) for x in t]
data4 = [STERBE_RISIKO for _ in t]
plt.rcParams['figure.figsize'] = [12, 8]
plt.rcParams['figure.dpi'] = 100 # 200 e.g. is really fine, but slower

fig, ax1 = plt.subplots()
color = 'tab:red'
ax1.set_xlabel('Zeit')
ax1.set_ylabel('Infektionen, Tote', color=color)
ax1.plot(t, data1, color=color)
ax1.plot(t, data2, color='tab:green')
ax1.tick_params(axis='y', labelcolor=color)
ax1.set_ylim([0,70])

ax2 = ax1.twinx()  # instantiate a second axes that shares the same x-axis
color = 'tab:blue'
ax2.set_ylabel('beobachtete, tatschliche Fallsterblichkeit', color=color)  # we already handled the x-label with ax1
ax2.plot(t, data3, color=color)
ax2.plot(t, data4, color="tab:grey")
ax2.tick_params(axis='y', labelcolor=color)
ax2.set_ylim(0,1.1)

fig.tight_layout()  # otherwise the right y-label is slightly clipped
plt.show()
